<script>

const USER = "vendomatica";
const PASS = "Vend321";

const data = [
{codigo:"3334803",nombre:"FILTRO CORRIENTE 5 AMPER PN 3030 - AMS",stock:2},
{codigo:"3339123",nombre:"BOMBA PN 098760 220V/50HZ",stock:0},
{codigo:"3339382",nombre:"FILTRO RED PN OV3512 KORO",stock:3}
];

function login(){

let u=document.getElementById('user').value;
let p=document.getElementById('pass').value;

if(u===USER && p===PASS){

document.getElementById('login').classList.add('hidden');
document.getElementById('app').classList.remove('hidden');

}else{
document.getElementById('msg').innerText="Usuario o contraseña incorrectos";
document.getElementById('msg').style.color="red";
}

}

function render(q=""){

const lista=document.getElementById('lista');
lista.innerHTML="";

const resultados=data.filter(x=>
x.nombre.toLowerCase().includes(q) ||
x.codigo.includes(q)
);

resultados.forEach(x=>{

let div=document.createElement('div');

div.className='item';

div.innerHTML=`
<b>${x.nombre}</b><br>
Código: ${x.codigo}<br>
${x.stock>0 ?
"<span class='ok'>🟢 STOCK DISPONIBLE</span>" :
"<span class='no'>🔴 SIN STOCK</span>"}
`;

div.onclick=()=>mostrar(x);

lista.appendChild(div);

});

}

function mostrar(x){

document.getElementById('nombre').innerText=x.nombre;
document.getElementById('codigo').innerText=x.codigo;
document.getElementById('stock').innerText=x.stock;

}

function copiarCodigo(){

let codigo=document.getElementById('codigo').innerText;

navigator.clipboard.writeText(codigo);

alert("Código copiado: " + codigo);

}

document.addEventListener('input',(e)=>{

if(e.target.id==='buscar'){
render(e.target.value.toLowerCase());
}

});

render();

if('serviceWorker' in navigator){
navigator.serviceWorker.register('service-worker.js');
}
